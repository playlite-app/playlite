1//! Módulo de gerenciamento de lista de desejos (wishlist).
2//!
3//! Implementa funcionalidades para rastreamento de jogos desejados,
4//! incluindo busca, adição, remoção e monitoramento de preços.
5//!
6//! # Funcionalidades Principais
7//! - Busca de jogos na loja Steam
8//! - Adição/remoção de itens da wishlist
9//! - Busca de preços
10//! - Atualização de preços via Steam Store API
11//! - Auto-healing de Steam App IDs faltantes
12//!
13//! # Monitoramento de Preços
14//! A função `refresh_prices` atualiza:
15//! - Preços atuais em BRL
16//! - Status de promoção (on_sale)
17//! - URLs diretas para loja
18

28/// Busca jogos na loja Steam por termo de pesquisa.
29///
30/// Retorna lista de resultados da Steam Store para seleção pelo usuário.
31/// Usado no modal de "Adicionar à Wishlist".
32///
33/// # Retorna
34/// * `Ok(Vec<StoreSearchItem>)` - Lista de jogos encontrados (vazia se nenhum)
35/// * `Err(String)` - Erro na comunicação com Steam Store API
36///
37/// # Exemplo de Uso
38/// ```typescript,ignore
39/// const results = await invoke('search_wishlist_game', {
40///     query: 'cyberpunk'
41/// });
42///
43/// // Mostrar resultados em modal de seleção
44/// results.forEach(game => {
45///     console.log(`${game.name} (ID: ${game.id})`);
46/// });
47/// ```
48///
49/// # Nota
50/// Esta é uma busca direta na Store, não requer autenticação.
51/// Retorna jogos de todas as regiões, mas priorizados para região BR.
52#[tauri::command]
53pub async fn search_wishlist_game(query: String) -> Result<Vec<StoreSearchItem>, String> {
54    steam::search_store(&query).await
55}

57/// Adiciona um jogo à lista de desejos.
58///
59/// Insere ou atualiza um jogo na wishlist com informações básicas.
60/// Usa `INSERT OR REPLACE` para permitir re-adicionar jogos removidos.
61#[tauri::command]
62pub fn add_to_wishlist(
63    state: State<AppState>,
64    id: String,
65    name: String,
66    cover_url: Option<String>,
67    store_url: Option<String>,
68    current_price: Option<f64>,
69    steam_app_id: Option<i32>,
70) -> Result<String, String> {
71    info!(
72        "Tentando adicionar à Wishlist: ID={}, Nome={}, SteamID={:?}",
73        id, name, steam_app_id
74    );
75
76    let conn = state.library_db.lock().map_err(|e| {
77        error!("Erro de Mutex na Wishlist: {}", e);
78        "Falha interna ao acessar banco".to_string()
79    })?;
80
81    // Tenta inserir e loga o resultado exato
82    match conn.execute(
83        "INSERT OR REPLACE INTO wishlist (id, name, cover_url, store_url, current_price, steam_app_id, added_at)
84         VALUES (?1, ?2, ?3, ?4, ?5, ?6, CURRENT_TIMESTAMP)",
85        params![id, name, cover_url, store_url, current_price, steam_app_id],
86    ) {
87        Ok(_) => {
88            info!("Sucesso: Jogo {} adicionado à wishlist.", name);
89            Ok("Jogo adicionado à lista de desejos!".to_string())
90        },
91        Err(e) => {
92            error!("Erro SQL ao adicionar {}: {:?}", name, e);
93            Err(format!("Erro de banco de dados: {}", e))
94        }
95    }
96}

98/// Remove um jogo da lista de desejos.
99///
100/// Remove o jogo identificado pelo ID fornecido da wishlist.
101#[tauri::command]
102pub fn remove_from_wishlist(state: State<AppState>, id: String) -> Result<String, String> {
103    let conn = state
104        .library_db
105        .lock()
106        .map_err(|_| "Falha ao bloquear mutex")?;
107
108    conn.execute("DELETE FROM wishlist WHERE id = ?1", params![id])
109        .map_err(|e| e.to_string())?;
110
111    Ok("Jogo removido da lista de desejos.".to_string())
112}

114/// Recupera todos os jogos da lista de desejos.
115///
116/// Retorna a wishlist completa ordenada por data de adição (mais recentes primeiro).
117#[tauri::command]
118pub fn get_wishlist(state: State<AppState>) -> Result<Vec<WishlistGame>, String> {
119    let conn = state
120        .library_db
121        .lock()
122        .map_err(|_| "Falha ao bloquear mutex")?;
123
124    let mut stmt = conn
125        .prepare("SELECT id, name, cover_url, store_url, current_price, lowest_price, on_sale, localized_price, localized_currency, steam_app_id, added_at FROM wishlist ORDER BY added_at DESC")
126        .map_err(|e| e.to_string())?;
127
128    let games_iter = stmt
129        .query_map([], |row| {
130            Ok(WishlistGame {
131                id: row.get(0)?,
132                name: row.get(1)?,
133                cover_url: row.get(2)?,
134                store_url: row.get(3)?,
135                current_price: row.get(4)?,
136                lowest_price: row.get(5)?,
137                on_sale: row.get(6)?,
138                localized_price: row.get(7)?,
139                localized_currency: row.get(8)?,
140                steam_app_id: row.get(9)?,
141                added_at: row.get(10)?,
142            })
143        })
144        .map_err(|e| e.to_string())?;
145
146    let mut games = Vec::new();
147    for game in games_iter {
148        games.push(game.map_err(|e| e.to_string())?);
149    }
150
151    Ok(games)
152}

154/// Verifica se um jogo está na lista de desejos.
155///
156/// Consulta rápida para verificar presença na wishlist, útil para
157/// atualizar UI (ex: mudar ícone de "adicionar" para "remover").
158#[tauri::command]
159pub fn check_wishlist_status(state: State<AppState>, id: String) -> Result<bool, String> {
160    let conn = state
161        .library_db
162        .lock()
163        .map_err(|_| "Falha ao bloquear mutex")?;
164
165    let count: i32 = conn
166        .query_row(
167            "SELECT COUNT(1) FROM wishlist WHERE id = ?1",
168            params![id],
169            |row| row.get(0),
170        )
171        .unwrap_or(0);
172
173    Ok(count > 0)
174}

176/// Atualiza preços dos jogos da wishlist.
177///
178/// Operação em lote que consulta a Steam Store API para cada jogo
179/// e atualiza informações de preço, desconto e URLs.
180///
181/// # Processo
182/// 1. Busca todos os jogos da wishlist
183/// 2. Para cada jogo:
184///    - Verifica se tem Steam App ID
185///    - Se não tem, tenta descobrir via busca (auto-healing)
186///    - Consulta preço atual na Steam Store API
187///    - Atualiza banco com novos dados
188///    - Aguarda 500ms (rate limiting)
189///
190/// # Retorna
191/// * `Ok(String)` - Mensagem com contador de atualizações
192/// * `Err(String)` - Erro crítico de banco ou sistema
193///
194/// # Exemplo de Uso
195/// ```typescript,ignore
196/// // Iniciar atualização (operação longa)
197/// const result = await invoke('refresh_prices');
198/// console.log(result); // "Preços atualizados: 23/25"
199///
200/// // Recarregar wishlist para mostrar novos preços
201/// const updated = await invoke('get_wishlist');
202/// ```
203///
204/// # Nota
205/// - Operação naturalmente lenta devido ao rate limiting (500ms) obrigatório.
206/// - Erros individuais não interrompem o processo.
207/// - Jogos sem Steam App ID que não são encontrados na busca são ignorados.
208#[tauri::command]
209pub async fn refresh_prices(state: State<'_, AppState>) -> Result<String, String> {
210    // Busca dados básicos do banco
211    let games: Vec<(String, Option<i32>, String)> = {
212        let conn = state
213            .library_db
214            .lock()
215            .map_err(|_| "Falha ao bloquear mutex")?;
216        let mut stmt = conn
217            .prepare("SELECT id, steam_app_id, name FROM wishlist")
218            .map_err(|e| e.to_string())?;
219
220        let rows = stmt
221            .query_map([], |row| Ok((row.get(0)?, row.get(1)?, row.get(2)?)))
222            .map_err(|e| e.to_string())?
223            .collect::<Result<Vec<_>, _>>()
224            .map_err(|e| e.to_string())?;
225        rows
226    };
227
228    let total = games.len();
229    let mut updated_count = 0;
230
231    for (id, steam_app_id, name) in games {
232        let mut current_app_id = steam_app_id;
233
234        // Se não tem AppID, tenta descobrir pelo nome (Auto-healing)
235        if current_app_id.is_none() {
236            // Nota: search_store retorna lista, pegamos o primeiro para auto-healing
237            if let Ok(results) = steam::search_store(&name).await {
238                if let Some(first) = results.first() {
239                    current_app_id = Some(first.id as i32);
240                    let conn = state.library_db.lock().map_err(|_| "Falha mutex")?;
241                    let _ = conn.execute(
242                        "UPDATE wishlist SET steam_app_id = ?1 WHERE id = ?2",
243                        params![current_app_id, &id],
244                    );
245                }
246            }
247        }
248
249        // Busca Preço na Steam
250        if let Some(app_id_val) = current_app_id {
251            match steam::fetch_price(app_id_val as u32).await {
252                Ok(Some(price)) => {
253                    let conn = state.library_db.lock().map_err(|_| "Falha mutex")?;
254                    let on_sale = price.discount_percent > 0;
255
256                    // URL da loja Steam para o botão "Ir para Loja"
257                    let store_url = format!("https://store.steampowered.com/app/{}/", app_id_val);
258
259                    // Atualiza BRL
260                    let _ = conn.execute(
261                        "UPDATE wishlist
262                            SET localized_price = ?1, localized_currency = ?2,
263                                on_sale = ?3, store_url = ?4,
264                                lowest_price = MIN(IFNULL(lowest_price, 9999), ?1)
265                            WHERE id = ?5",
266                        params![price.final_price, price.currency, on_sale, store_url, id],
267                    );
268                    updated_count += 1;
269                }
270                Ok(None) => {
271                    println!("Jogo indisponível na loja BR: {}", name);
272                }
273                Err(_) => {}
274            }
275        }
276
277        sleep(Duration::from_millis(500)).await;
278    }
279
280    Ok(format!("Preços atualizados: {}/{}", updated_count, total))
281}
