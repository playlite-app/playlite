1//! Módulo de gerenciamento do banco de dados da aplicação.
2//!
3//! Gerencia a criação e inicialização do banco SQLite para a biblioteca de jogos e wishlist,
4//! além do armazenamento seguro de secrets (API keys, tokens) com criptografia.
5//!
6//! # Bancos de Dados
7//! - library.db: armazena a biblioteca de jogos e wishlist do usuário.
8//! - secrets.db: armazena secrets encriptados com AES-256-GCM.
9
10use crate::constants::{DB_FILENAME_LIBRARY, DB_FILENAME_SECRETS, DB_JOURNAL_MODE};
11use crate::security;
12use rusqlite::{params, Connection};
13use std::sync::Mutex;
14use tauri::State;
15use tauri::{AppHandle, Manager};
16
17/// Define o estado global da aplicação com ambas as conexões
18pub struct AppState {
19    pub library_db: Mutex<Connection>,
20    pub secrets_db: Mutex<Connection>,
21}
22
23// === INICIALIZAÇÃO CENTRALIZADA ===
24
25/// Inicializa ambos os bancos de dados e retorna o estado da aplicação
26///
27/// # Erros
28/// - Se não conseguir criar os diretórios
29/// - Se não conseguir abrir as conexões
30/// - Se falhar ao configurar WAL mode
31pub fn initialize_databases(app: &AppHandle) -> Result<AppState, String> {
32    let app_data_dir = app
33        .path()
34        .app_data_dir()
35        .map_err(|e| format!("Falha ao obter app_data_dir: {}", e))?;
36
37    std::fs::create_dir_all(&app_data_dir)
38        .map_err(|e| format!("Falha ao criar diretório: {}", e))?;
39
40    // Conexão para library.db
41    let library_path = app_data_dir.join(DB_FILENAME_LIBRARY);
42    let library_conn = Connection::open(&library_path)
43        .map_err(|e| format!("Erro ao abrir {}: {}", DB_FILENAME_LIBRARY, e))?;
44
45    library_conn
46        .pragma_update(None, "journal_mode", DB_JOURNAL_MODE)
47        .map_err(|e| format!("Erro ao configurar WAL no library.db: {}", e))?;
48
49    tracing::info!(
50        "Banco {} inicializado em: {:?}",
51        DB_FILENAME_LIBRARY,
52        library_path
53    );
54
55    // Conexão para secrets.db
56    let secrets_path = app_data_dir.join(DB_FILENAME_SECRETS);
57    let secrets_conn = Connection::open(&secrets_path)
58        .map_err(|e| format!("Erro ao abrir {}: {}", DB_FILENAME_SECRETS, e))?;
59
60    secrets_conn
61        .pragma_update(None, "journal_mode", DB_JOURNAL_MODE)
62        .map_err(|e| format!("Erro ao configurar WAL no secrets.db: {}", e))?;
63
64    tracing::info!(
65        "Banco {} inicializado em: {:?}",
66        DB_FILENAME_SECRETS,
67        secrets_path
68    );
69
70    Ok(AppState {
71        library_db: Mutex::new(library_conn),
72        secrets_db: Mutex::new(secrets_conn),
73    })
74}

76// === BANCO DE DADOS PARA GERENCIAMENTO DE JOGOS (library.db) ===
77
78/// Inicializa o banco de dados de gerenciamento de jogos.
79///
80/// Cria as tabelas `games` e `wishlist` se não existirem, e índices otimizados para consultas frequentes.
81///
82/// # Erros
83/// - Se falhar ao bloquear o mutex do banco de dados.
84/// - Se falhar ao executar comandos SQL.
85#[tauri::command]
86pub fn init_db(state: State<AppState>) -> Result<String, String> {
87    let conn = state
88        .library_db
89        .lock()
90        .map_err(|_| "Falha ao bloquear mutex do library_db")?;
91
92    // === TABELAS BÁSICAS ===
93
94    // Tabela da Biblioteca de Jogos
95    conn.execute(
96        "CREATE TABLE IF NOT EXISTS games (
97            id TEXT PRIMARY KEY,
98            name TEXT NOT NULL,
99            genre TEXT,
100            platform TEXT,
101            cover_url TEXT,
102            playtime INTEGER DEFAULT 0,
103            rating INTEGER,
104            favorite BOOLEAN DEFAULT FALSE
105        )",
106        [],
107    )
108    .map_err(|e| e.to_string())?;
109
110    // Tabela da Lista de Desejos
111    conn.execute(
112        "CREATE TABLE IF NOT EXISTS wishlist (
113            id TEXT PRIMARY KEY,
114            name TEXT NOT NULL,
115            cover_url TEXT,
116            store_url TEXT,
117            current_price REAL,
118            lowest_price REAL,
119            on_sale BOOLEAN DEFAULT 0,
120            added_at DATETIME DEFAULT CURRENT_TIMESTAMP,
121            localized_price REAL,
122            localized_currency TEXT,
123            steam_app_id INTEGER
124        )",
125        [],
126    )
127    .map_err(|e| e.to_string())?;
128
129    // === ÍNDICES OTIMIZADOS ===
130
131    conn.execute(
132        "CREATE INDEX IF NOT EXISTS idx_favorite ON games(favorite)",
133        [],
134    )
135    .map_err(|e| e.to_string())?;
136
137    conn.execute(
138        "CREATE INDEX IF NOT EXISTS idx_name ON games(name COLLATE NOCASE)",
139        [],
140    )
141    .map_err(|e| e.to_string())?;
142
143    conn.execute(
144        "CREATE INDEX IF NOT EXISTS idx_platform ON games(platform)",
145        [],
146    )
147    .map_err(|e| e.to_string())?;
148
149    conn.execute(
150        "CREATE INDEX IF NOT EXISTS idx_wishlist_added ON wishlist(added_at)",
151        [],
152    )
153    .map_err(|e| e.to_string())?;
154
155    Ok("Banco de jogos inicializado com sucesso!".to_string())
156}

158// === BANCO DE DADOS PARA GERENCIAMENTO DE API KEYS (secrets.db) ===
159
160/// Obtém conexão com o banco de secrets a partir do AppState.
161///
162/// Cria automaticamente a tabela `encrypted_keys` se não existir.
163///
164/// # Erros
165/// - Se falhar ao bloquear o mutex do banco de dados.
166/// - Se falhar ao executar comandos SQL.
167fn get_secrets_connection<'a>(
168    state: &'a State<AppState>,
169) -> Result<std::sync::MutexGuard<'a, Connection>, String> {
170    let conn = state
171        .secrets_db
172        .lock()
173        .map_err(|_| "Falha ao bloquear mutex do secrets_db".to_string())?;
174
175    // Garante que a tabela existe
176    conn.execute(
177        r#"
178        CREATE TABLE IF NOT EXISTS encrypted_keys (
179            key TEXT PRIMARY KEY,
180            value TEXT NOT NULL
181        )
182        "#,
183        [],
184    )
185    .map_err(|e: rusqlite::Error| e.to_string())?;
186
187    Ok(conn)
188}

190/// Salva um secret encriptado no banco.
191///
192/// Se a chave já existir, o valor é substituído (upsert).
193/// O valor é encriptado antes de ser armazenado.
194pub fn set_secret(app: &AppHandle, key_name: &str, value: &str) -> Result<(), String> {
195    // Obtém o AppState
196    let state: tauri::State<AppState> = app.state();
197    let conn = get_secrets_connection(&state)?;
198
199    let encrypted = security::encrypt(value);
200
201    conn.execute(
202        "INSERT OR REPLACE INTO encrypted_keys (key, value) VALUES (?1, ?2)",
203        params![key_name, encrypted],
204    )
205    .map_err(|e| e.to_string())?;
206
207    Ok(())
208}

210/// Recupera e decripta um secret do banco.
211///
212/// Se a chave não existir, retorna string vazia ao invés de erro.
213pub fn get_secret(app: &AppHandle, key_name: &str) -> Result<String, String> {
214    let state: tauri::State<AppState> = app.state();
215    let conn = get_secrets_connection(&state)?;
216
217    let result: Result<String, rusqlite::Error> = conn.query_row(
218        "SELECT value FROM encrypted_keys WHERE key = ?1",
219        params![key_name],
220        |row| row.get::<_, String>(0),
221    );
222
223    match result {
224        Ok(encrypted) => {
225            let decrypted = security::decrypt(&encrypted)?;
226            Ok(decrypted)
227        }
228        Err(rusqlite::Error::QueryReturnedNoRows) => Ok(String::new()),
229        Err(e) => Err(e.to_string()),
230    }
231}

233/// Remove um secret do banco permanentemente.
234pub fn delete_secret(app: &AppHandle, key_name: &str) -> Result<(), String> {
235    let state: tauri::State<AppState> = app.state();
236    let conn = get_secrets_connection(&state)?;
237
238    conn.execute(
239        "DELETE FROM encrypted_keys WHERE key = ?1",
240        params![key_name],
241    )
242    .map_err(|e: rusqlite::Error| e.to_string())?;
243
244    Ok(())
245}
