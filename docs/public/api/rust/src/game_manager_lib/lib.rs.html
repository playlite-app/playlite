1//! # Playlite - Game Manager Library
2//!
3//! Fornece funcionalidades para:
4//! - Importar jogos do Steam
5//! - Gerenciar biblioteca pessoal
6//! - Wishlist com tracking de preços
7//! - Armazenamento seguro de API keys
8//! - Busca de jogos em tendência (RAWG API)
9
10pub mod commands;
11mod constants;
12mod database;
13pub mod models;
14mod security;
15pub mod services;
16pub mod utils;
17
18use crate::utils::logger;
19use tauri::Manager;
20
21#[cfg_attr(mobile, tauri::mobile_entry_point)]
22pub fn run() {
23    tauri::Builder::default()
24        .plugin(tauri_plugin_shell::init())
25        .plugin(tauri_plugin_opener::init())
26        .plugin(tauri_plugin_store::Builder::default().build())
27        .plugin(tauri_plugin_dialog::init())
28        .plugin(tauri_plugin_machine_uid::init())
29        .setup(|app| {
30            let app_handle = app.handle();
31
32            // === LOGGING ===
33            let log_dir = app_handle
34                .path()
35                .app_log_dir()
36                .expect("Falha ao pegar pasta de log");
37
38            std::fs::create_dir_all(&log_dir).expect("Falha ao criar pasta de dev_logs");
39
40            let _guard = logger::init_logging(log_dir.clone());
41            app.manage(_guard);
42
43            tracing::info!("Aplicação iniciada! Logs em: {:?}", log_dir);
44
45            // === SEGURANÇA ===
46            security::init_security(app_handle).expect("Falha ao inicializar sistema de segurança");
47
48            tracing::info!("Sistema de segurança inicializado");
49
50            // === BANCOS DE DADOS ===
51            let db_state = database::initialize_databases(app_handle)
52                .expect("Falha ao inicializar bancos de dados");
53
54            app.manage(db_state);
55
56            tracing::info!("Bancos de dados inicializados com sucesso");
57
58            Ok(())
59        })
60        .invoke_handler(tauri::generate_handler![
61            // Comando de Inicialização do Banco de Dados
62            database::init_db,
63            // Comandos de Jogos (CRUD)
64            commands::games::add_game,
65            commands::games::get_games,
66            commands::games::toggle_favorite,
67            commands::games::delete_game,
68            commands::games::update_game,
69            // Comandos da Lista de Desejos
70            commands::wishlist::search_wishlist_game,
71            commands::wishlist::add_to_wishlist,
72            commands::wishlist::get_wishlist,
73            commands::wishlist::remove_from_wishlist,
74            commands::wishlist::check_wishlist_status,
75            commands::wishlist::refresh_prices,
76            // Comandos de Integração (Steam/RAWG)
77            commands::integrations::import_steam_library,
78            commands::integrations::enrich_library,
79            commands::integrations::get_trending_games,
80            commands::integrations::get_upcoming_games,
81            commands::integrations::fetch_game_details,
82            // Comandos de Configuração (Secrets)
83            commands::settings::set_secret,
84            commands::settings::get_secret,
85            commands::settings::delete_secret,
86            commands::settings::list_secrets,
87            commands::settings::get_secrets,
88            commands::settings::set_secrets,
89            // Comandos de Backup e Restauração
90            commands::backup::export_database,
91            commands::backup::import_database,
92            // Comando de Recomendação
93            commands::recommendations::get_user_profile
94        ])
95        .run(tauri::generate_context!())
96        .expect("error while running tauri application");
97}
