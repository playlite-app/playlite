1//! Módulo de sistema de recomendação personalizado.
2//!
3//! Implementa algoritmo de análise de perfil baseado em múltiplos fatores:
4//! tempo de jogo, avaliações, favoritos e gêneros preferidos.
5//!
6//! O sistema calcula scores ponderados para cada gênero e gera um perfil
7//! do usuário que pode ser usado para ranquear e recomendar novos jogos.
8
9use crate::models::{Game, GenreScore, UserProfile};
10use std::collections::HashMap;
11
12// === Configuração de Pesos do Algoritmo ===
13
14/// Pontos atribuídos por hora jogada.
15///
16/// Cada hora de jogo adiciona 2 pontos ao score do jogo, limitado a 100 horas para evitar outliers.
17const WEIGHT_PLAYTIME_HOUR: f32 = 2.0;
18
19/// Pontos de bônus para jogos marcados como favorito.
20///
21/// Jogos favoritos recebem 50 pontos adicionais, indicando preferência do usuário.
22const WEIGHT_FAVORITE: f32 = 50.0;
23
24/// Pontos atribuídos por estrela de avaliação.
25///
26/// Sistema de 5 estrelas onde cada estrela vale 10 pontos.
27const WEIGHT_RATING_STAR: f32 = 10.0;
28
29/// Fator de decaimento temporal para jogos antigos (não implementado).
30///
31/// Planejado para reduzir gradualmente a influência de títulos jogados há muito tempo.
32/// Valor de 0,95 - para 5% de redução por período definido.
33#[allow(dead_code)]
34const DECAY_FACTOR: f32 = 0.95;
35
36/// Calcula o perfil de preferências do usuário.
37///
38/// Analisa a biblioteca de jogos do usuário e gera um perfil baseado em:
39/// - **Tempo de jogo**: Maior peso para jogos mais jogados (até 100h)
40/// - **Favoritos**: Bônus significativo para jogos marcados
41/// - **Avaliações**: Score baseado em estrelas atribuídas
42/// - **Gêneros**: Distribuição dos scores pelos gêneros dos jogos
43///
44/// # Algoritmo
45///
46/// Para cada jogo:
47/// 1. Calcula score base: `(horas * 2) + (favorito * 50) + (estrelas * 10)`
48/// 2. Distribui esse score entre todos os gêneros do jogo
49/// 3. Acumula scores e contadores por gênero
50///
51/// # Parâmetros
52/// * `games` - Slice de jogos da biblioteca do usuário
53///
54/// # Retorna
55/// `UserProfile` contendo:
56/// - `top_genres`: Gêneros ordenados por score (maior → menor)
57/// - `total_playtime`: Soma total de minutos jogados
58/// - `total_games`: Quantidade de jogos analisados
59///
60/// # Exemplo
61/// ```rust
62/// let profile = calculate_user_profile(&user_games);
63///
64/// println!("Gênero favorito: {}", profile.top_genres[0].name);
65/// println!("Score do gênero: {}", profile.top_genres[0].score);
66/// println!("Total de horas: {}", profile.total_playtime / 60);
67/// ```
68pub fn calculate_user_profile(games: &[Game]) -> UserProfile {
69    let mut genre_scores: HashMap<String, (f32, i32)> = HashMap::new();
70    let mut total_playtime = 0;
71
72    for game in games {
73        total_playtime += game.playtime;
74
75        // Calcular o Score Base do Jogo (Game Score)
76        let mut game_score = 0.0;
77
78        // Fator Tempo de Jogo (limitado a 100h para não distorcer demais)
79        let hours = (game.playtime as f32 / 60.0).min(100.0);
80        game_score += hours * WEIGHT_PLAYTIME_HOUR;
81
82        // Fator Favorito
83        if game.favorite {
84            game_score += WEIGHT_FAVORITE;
85        }
86
87        // Fator Rating (Se tiver avaliação)
88        if let Some(rating) = game.rating {
89            game_score += (rating as f32) * WEIGHT_RATING_STAR;
90        }
91
92        // Distribuir o Score para os Gêneros do Jogo
93        if let Some(genre_str) = &game.genre {
94            let genres: Vec<&str> = genre_str.split(',').map(|s| s.trim()).collect();
95
96            for genre in genres {
97                if genre.is_empty() || genre == "Desconhecido" {
98                    continue;
99                }
100
101                let entry = genre_scores.entry(genre.to_string()).or_insert((0.0, 0));
102                entry.0 += game_score; // Soma pontuação
103                entry.1 += 1; // Conta ocorrência
104            }
105        }
106    }
107
108    // Converter HashMap para Vetor Ordenado
109    let mut top_genres: Vec<GenreScore> = genre_scores
110        .into_iter()
111        .map(|(name, (score, count))| GenreScore {
112            name,
113            score,
114            game_count: count,
115        })
116        .collect();
117
118    // Ordenar do maior score para o menor
119    top_genres.sort_by(|a, b| {
120        b.score
121            .partial_cmp(&a.score)
122            .unwrap_or(std::cmp::Ordering::Equal)
123    });
124
125    UserProfile {
126        top_genres,
127        total_playtime,
128        total_games: games.len() as i32,
129    }
130}
